drop table tabla_temporal;
create table tabla_temporal(id_empleado number(4) , mensage varchar2(50));

declare
    sueldo employee.salary %type;
    codigo employee.employee_id % type;
begin
    select salary into sueldo from employee where employee_id = &codigo;
    if(sueldo >1000) then
        insert into tabla_temporal values(codigo,"Sueldo actualizado");
    end if;
end;
